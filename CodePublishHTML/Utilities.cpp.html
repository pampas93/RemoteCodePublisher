std::vector<std::string> StringHelper::split(const std::string& src)

-  {

  std::vector<std::string> accum;
  std::string temp;
  std::locale loc;
  size_t index = 0;
  do
  {
    while ((isspace(src[index], loc) || src[index] == ',') && src[index] != '\n')
    {
      ++index;
      if (temp.size() > 0)
      {
        accum.push_back(temp);
        temp.clear();
      }
    }
    if(src[index] != '\0')
      temp += src[index];
  } while (index++ < src.size());
  if(temp.size() > 0)
    accum.push_back(temp);
  return accum;

}
